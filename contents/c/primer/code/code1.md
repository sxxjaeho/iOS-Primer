 // 这是一个预编译指令，主要是为了保证下面使用 printf 函数

 #include <stdio.h>
 /*
 1、C语言源程序的后缀：.c 格式
  
 2、C语言源程序：
 1）有函数构成
 2）在一个程序中 ，有且只有一个主函数（主函数由系统调用）
 3）函数只有被调用的时候才执行
 4）如果没有主函数程序无法运行
 5）C语言中语句结束一定要有分号

 3、主函数的写法：
  int main(){
  	printf("hello world");
  	return 0;
  } 
  
 4、C语言程序的执行 
 1）command+r
 2）点击左上角的 三角符号 
 
 5、C语言程序执行流程
 写源文件
        编译
		---> .o 后缀的目标文件
			                链接（把目标文件和库函数整合生成一个 .out 可执行文件） 
							---> .out 后缀的可执行文件								
  */
   
 // 这是一个C语言的主函数
 int main(){
	printf("hello world \n");
  	return 0;// 表示main函数返回给系统一个 0
	         // 表示程序正常执行，并且已经退出 
 }
